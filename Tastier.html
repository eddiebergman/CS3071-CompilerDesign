Tastier Grammar

Tastier =     "program" ident "{" {VarDecl | ConstDecl} {ProcDecl} "}"


VarDecl =     Type ident [ ArrayDecl ] {"," ident [ ArrayDecl ] } ";"

ProcDecl =   "void" ident "(" ")"

             "{" {VarDecl | ConstDecl } {ProcDecl} Stat {Stat} "}"


Type =       "int" | "bool"


Stat =       ident (":=" Expr ";" | "(" ")" ";" | ArrayIndex ":=" Expr ';' ')

           | "if" Expr Stat ["else" Stat]

           | "while Expr Stat

           | "read" ident ";"

           | "write" Expr | string ";"

           | "writeln" Expr ";"

           | "{" {VarDecl} Stat {Stat} ”}”


Expr =       SimExpr [RelOp SimExpr [ TernaryOp ]]

SimExpr =     Term {AddOp Term}

Term =       Primary {MulOp Primary}

Primary = (ident [ ArrayIndex ] | number | "-" Primary  

             | "true" | "false" | "(" Expr ")")


AddOp =       "+" | "-"

MulOp =       "*" | "div" | "DIV" | "mod" | "MOD"

RelOp =       "=" | "<" | ">" | "|=" | "<=" | ">="

TernaryOp =       '?' Expr ':' Expr

ConstDecl =       "const" Type Ident "=" ConstExpr

ArrayDecl =       '[' ConstExpr ']' { '[' ConstExpr ']' }

ConstExpr =       (number | '-'number | "true" | "false" | Ident)

ArrayIndex =       '[' Expr ']' { '[' Expr ']' }

Switch =       'switch' Ident ‘{‘ { 'case' Primary ':' { Stat } [ 'break' ';' ] } [ 'default' ':' { Stat } [ 'break' ';'] ] '}'
